 "routeId": 1,
 "departureTime": "2026-03-10T18:00:00",
 "bookingOpenTime": "2026-03-10T15:00:00",
 "bookingCloseTime": "2026-03-10T17:45:00",

{
 "routeId": "hello",
 "price": "cheap"
}
